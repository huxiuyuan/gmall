### 配置nginx的配置文件

```shell
# 进入nginx容器
docker exec -it nginx bash
# 容器内安装vim
apt-get install vim
# 如果报错 E: Unable to locate package vim
apt-get update
# 再执行一次安装
apt-get install vim

# 编辑nginx配置文件
vim /etc/nginx/nginx.conf

# 配置文件内容 选择本机的固定ip配置
user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log notice;
pid        /var/run/nginx.pid;


events {
    worker_connections  1024;
}


http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile        on;
    #tcp_nopush     on;

    keepalive_timeout  65;

    #gzip  on;

    include /etc/nginx/conf.d/*.conf;

    server {
        listen		   80;
        server_name   api.gmall.com search.gmall.com www.gmall.com gmall.com;

        proxy_set_header Host $host;

        location / {
            proxy_pass http://192.168.247.1:8888;
        }
    }

    server {
        listen		   80;
        server_name   item.gmall.com;

        proxy_set_header Host $host;

        location / {
            root /opt/html;
            if (!-f $request_filename) {
              proxy_pass http://192.168.247.1:8888;
            }
        }
    }

    server {
        listen		   80;
        server_name   manager.gmall.com;

        location / {
            proxy_pass http://192.168.247.1:1000;
        }
    }

    server {
        listen		   80;
        server_name  static.gmall.com;

        location / {
            root /opt/static/;
        }
    }
}

# 保存文件并退出
Esc :wq Enter

# 重启
docker restart nginx
```
